// --- Local Storage ---
function loadData() {
  const s = localStorage.getItem('nx-sources');
  const c = localStorage.getItem('nx-config');
  const n = localStorage.getItem('nx-name');
  if (s) state.sources = JSON.parse(s);
  if (c) state.settings = { ...state.settings, ...JSON.parse(c) };
  if (n) state.notebookName = n;
}
function saveData() {
  localStorage.setItem('nx-sources', JSON.stringify(state.sources));
  localStorage.setItem('nx-config', JSON.stringify(state.settings));
  localStorage.setItem('nx-name', state.notebookName);
  renderSources();
}

// --- Source Management ---
function renderSources() {
  const list = document.getElementById('source-list');
  list.innerHTML = state.sources.map(s => `

function addSourceObj(type, name, content) {
  state.sources.push({ id: Date.now().toString(), type, name, content });
  saveData();
  closeModal('add-modal');
}

function deleteSource(id) {
  state.sources = state.sources.filter(s => s.id !== id);
  saveData();
}

// --- Source Extractors ---
function addText() {
  const title = document.getElementById('txt-name').value || 'Text Note';
  const body = document.getElementById('txt-body').value;
  if (body) addSourceObj('text', title, body);
}

async function addURL() {
  const url = document.getElementById('url-input').value;
  const status = document.getElementById('url-status');
  if(!url) return;
  status.textContent = "Fetching...";
  try {
    const res = await fetch(`https://api.allorigins.win/get?url=${encodeURIComponent(url)}`);
    const data = await res.json();
    const div = document.createElement('div');
    div.innerHTML = data.contents;
    const text = div.textContent.replace(/\s+/g, ' ').trim();
    addSourceObj('url', new URL(url).hostname, text);
    status.textContent = "";
    document.getElementById('url-input').value = "";
  } catch(e) { status.textContent = "Failed to fetch URL."; }
}

async function handleFiles(e) {
  const files = e.target.files;
  const status = document.getElementById('file-status');
  for (let f of files) {
    status.textContent = `Processing ${f.name}...`;
    if (f.type === 'application/pdf' || f.name.endsWith('.pdf')) {
      const text = await extractPDF(f);
      addSourceObj('pdf', f.name, text);
    } else {
      const text = await f.text();
      addSourceObj('text', f.name, text);
    }
  }
  status.textContent = "";
  e.target.value = "";
}

async function extractPDF(file) {
  const arrayBuffer = await file.arrayBuffer();
  const pdf = await pdfjsLib.getDocument(arrayBuffer).promise;
  let fullText = "";
  for (let i = 1; i <= pdf.numPages; i++) {
    const page = await pdf.getPage(i);
    const textContent = await page.getTextContent();
    fullText += textContent.items.map(item => item.str).join(' ') + "\n";
  }
  return fullText;
}

async function addGitHub() {
  const repo = document.getElementById('git-repo').value.trim();
  const branch = document.getElementById('git-branch').value.trim() || 'main';
  const status = document.getElementById('git-status');
  if (!repo) return;
  status.textContent = "Fetching repo tree...";
  try {
    const res = await fetch(`https://api.github.com/repos/${repo}/git/trees/${branch}?recursive=1`);
    const data = await res.json();
    const files = data.tree.filter(f => f.type === 'blob' && /\.(js|ts|py|html|css|json|md|txt)$/.test(f.path)).slice(0, 30);
    
    let content = `Repo: ${repo}\n\n`;
    for (let f of files) {
      status.textContent = `Fetching ${f.path}...`;
      const fr = await fetch(`https://raw.githubusercontent.com/${repo}/${branch}/${f.path}`);
      if (fr.ok) content += `\n--- ${f.path} ---\n${await fr.text()}\n`;
    }
    addSourceObj('git', repo, content);
    status.textContent = "";
  } catch (e) { status.textContent = "Error fetching repo."; }
}

async function sendMessage() {
  const input = document.getElementById('chat-input');
  const text = input.value.trim();
  if (!text) return;
  
  if (!state.settings.key) {
    alert("Please add your API key in Settings (⚙️) first.");
    return;
  }

  document.getElementById('guide-view').style.display = 'none';
  const chatView = document.getElementById('chat-view');
  chatView.style.display = 'flex';
  
  input.value = "";
  input.style.height = 'auto';
  
  // Add User Message
  addMessageToUI('user', text);
  state.messages.push({ role: 'user', content: text });
  
  // Context Building
  const context = state.sources.map(s => `--- Source: ${s.name} ---\n${s.content.substring(0, 10000)}`).join('\n\n');
  const systemPrompt = `You are an AI assistant in a tool similar to NotebookLM. Answer the user strictly based on the provided sources. If the answer is not in the sources, say so. Use Markdown for formatting.\n\nSources:\n${context}`;

  const btn = document.getElementById('send-btn');
  btn.disabled = true;
  
  const loadingId = 'loading-' + Date.now();
  addMessageToUI('ai', '...', loadingId);

  try {
    const reply = await fetchAI(systemPrompt, state.messages);
    document.getElementById(loadingId).innerHTML = marked.parse(reply);
    state.messages.push({ role: 'assistant', content: reply });
  } catch (e) {
    document.getElementById(loadingId).innerHTML = `<span style="color:#f28b82">Error: ${e.message}</span>`;
  }
  
  btn.disabled = false;
  scrollToBottom();
}

function addMessageToUI(role, content, id = null) {
  const chatView = document.getElementById('chat-view');
  const div = document.createElement('div');
  div.className = `msg ${role}`;
  div.innerHTML = `
    <div class="msg-avatar">${role === 'user' ? '👤' : '✨'}</div>
    <div class="msg-content" ${id ? `id="${id}"` : ''}>${role === 'ai' ? (content === '...' ? content : marked.parse(content)) : content}</div>
  `;
  chatView.appendChild(div);
  scrollToBottom();
}

function scrollToBottom() {
  const contentArea = document.getElementById('content-area');
  contentArea.scrollTop = contentArea.scrollHeight;
}

async function fetchAI(system, msgs) {
  const { provider, model, key } = state.settings;
  
  if (provider === 'anthropic') {
    const res = await fetch('https://api.anthropic.com/v1/messages', {
      method: 'POST',
      headers: { 'x-api-key': key, 'anthropic-version': '2023-06-01', 'content-type': 'application/json', 'anthropic-dangerous-direct-browser-access':'true' },
      body: JSON.stringify({ model, max_tokens: 2000, system, messages: msgs })
    });
    const data = await res.json();
    if (!res.ok) throw new Error(data.error?.message || "Anthropic Error");
    return data.content[0].text;
  } else {
    const res = await fetch('https://openrouter.ai/api/v1/chat/completions', {
      method: 'POST',
      headers: { 'Authorization': `Bearer ${key}`, 'Content-Type': 'application/json' },
      body: JSON.stringify({ model, messages: [{ role: 'system', content: system }, ...msgs] })
    });
    const data = await res.json();
    if (!res.ok) throw new Error(data.error?.message || "OpenRouter Error");
    return data.choices[0].message.content;
  }
}

// --- Audio Overview (Native TTS) ---
async function playAudioOverview() {
  if (!state.sources.length) return alert("Add sources first to generate an overview.");
  if (!state.settings.key) return alert("Add API key to generate overview script.");
  
  const btn = document.getElementById('audio-btn');
  
  // If already speaking, stop it
  if (window.speechSynthesis.speaking) {
    window.speechSynthesis.cancel();
    btn.innerHTML = "▶ Play Overview";
    return;
  }

  btn.innerHTML = "⏳ Generating Script...";
  
  // Generate script
  const context = state.sources.map(s => s.content.substring(0, 5000)).join('\n');
  const prompt = `You are a podcast host. Summarize the following sources in a conversational, engaging, audio-friendly format. Keep it under 300 words. No markdown, just spoken text.\n\nSources:\n${context}`;
  
  try {
    const script = await fetchAI("You are a voice script writer.", [{role:'user', content:prompt}]);
    
    // Speak
    btn.innerHTML = "⏹ Stop Overview";
    const utterance = new SpeechSynthesisUtterance(script);
    utterance.rate = 1.05;
    utterance.pitch = 1;
    
    // Try to find a good voice (Google US English if available)
    const voices = window.speechSynthesis.getVoices();
    const goodVoice = voices.find(v => v.name.includes('Google') || v.name.includes('Premium'));
    if (goodVoice) utterance.voice = goodVoice;
    
    utterance.onend = () => { btn.innerHTML = "▶ Play Overview"; };
    window.speechSynthesis.speak(utterance);
    
  } catch (e) {
    alert("Error generating audio script: " + e.message);
    btn.innerHTML = "▶ Play Overview";
  }
}
